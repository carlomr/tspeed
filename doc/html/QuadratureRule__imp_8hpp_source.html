25  {
26  unsigned int n=N;

28  double xm=0.5*(b+a);
29  double xl=0.5*(b-a);

31  double z, z1, pi=M_PI, p1, p2, p3, pp;
32  double eps = std::numeric_limits<double>::epsilon();
33  //double eps = 1e-28;
34 
35  for(int i=1; i<=m; ++i){
36  z=cos(pi*(i-0.25)/(n+0.5));
37  do{
38  p1=1.0;
39  p2=0.0;
40  for(int j=1; j<=n; ++j)
41  {
42  p3=p2;
43  p2=p1;
44  p1=((2.0*j-1.0)*z*p2-(j-1.0)*p3)/j;
45  }
46  pp=n*(z*p1-p2)/(z*z-1.0);
47  z1=z;
48  z=z1-p1/pp;
49  }while(std::abs(z-z1)>eps);
50  x(i-1)=xm-xl*z;
51  x(n-i)=xm+xl*z;
52  //std::cout << n-i << ": " << x(n-i) << std::endl;
